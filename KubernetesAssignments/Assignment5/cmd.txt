# Create deployment object for metadata-service

apiVersion: apps/v1
kind: Deployment
metadata:
  name: metadata-service
  labels:
    app: metadata-service
spec:
  replicas: 2
  selector:
    matchLabels:
      app: metadata-service
  template:
    metadata:
      labels:
        app: metadata-service
    spec:
      containers:
      - name: metadata-service
        image: luckyganesh/metadata-service:v1-arm
        ports:
        - containerPort: 8080
