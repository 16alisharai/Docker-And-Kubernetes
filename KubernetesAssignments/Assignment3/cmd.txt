#Create NodePort service for MetadataService

Step-1 : Create deployment for metadata-service

-> vim deployment.yaml

apiVersion: apps/v1
kind: Deployment
metadata:
  name: metadata-service
  labels:
    app: metadata-service
spec:
  replicas: 2
  selector:
    matchLabels:
      app: metadata-service
  template:
    metadata:
      labels:
        app: metadata-service
    spec:
      containers:
      - name: metadata-service
        image: luckyganesh/metadata-service:v1-arm
        ports:
        - containerPort: 8080


-> kubectl create -f deployment.yaml

Step-2 : Check for the running pods and deployment
-> kubectl get deploy

Step-3 : Create service.yaml

-> vim service.yaml

apiVersion: v1
kind: Service
metadata:
  name: metadata-service
spec:
  type: NodePort
  selector:
    app: metadata-service
  ports:
    - port: 80
      targetPort: 8080
      nodePort: 30007
-> kubectl apply -f service.yaml
